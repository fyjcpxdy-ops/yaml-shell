选择其中支持的，在model:里面填写，例如  cortex-a57

```yaml
root@kubevirt1910b:/data/vm# cat vm.yaml 
apiVersion: kubevirt.io/v1
kind: VirtualMachine
metadata:
  name: ubuntu20-vm  # 自定义VM名
  namespace: default
  annotations:
    kubevirt.io/latest-observed-api-version: v1
    kubevirt.io/storage-observed-api-version: v1
spec:
  runStrategy: Manual  # 手动控制启动/停止，避免自动操作冲突
  template:
    metadata:
      labels:
        kubevirt.io/domain: ubuntu20-vm  # Kubevirt识别VM的默认标签，用于调度匹配
    spec:
      architecture: arm64  # 强制aarch64架构，匹配鲲鹏920
      domain:
        # 1. CPU模块（核心：QEMU 4.1.0支持的armv8-a）
        cpu:
          cores: 4
          sockets: 1
          threads: 1
          model:  cortex-a57  # 支持的aarch64模型，不可修改
        # 2. 机器类型模块（匹配节点标签）
        machine:
          type: virt-rhel9.4.0  # 与节点machine-type标签完全一致
        # 3. 设备模块（存储+网络+可选NPU直通）
        devices:
          # 存储设备（DataVolume+cloud-init）
          disks:
          - name: datavolumedisk  # 对应DataVolume存储
            disk:
              bus: virtio  # aarch64下高性能IO总线，必选
          - name: cloudinitdisk  # 云初始化配置
            disk:
              bus: virtio
          # 网络设备（NAT模式，简单稳定）
          interfaces:
          - name: default
            masquerade: {}  # 基于Pod的NAT网络，无需额外配置
          # 可选：昇腾910B NPU直通（需节点标签匹配）
          hostDevices:
          - deviceName: huawei.com/Ascend910  # 节点暴露的NPU资源名
            name: gpu0  # 自定义设备标识
            tag: gpu0   # 与设备标签对应（可选，用于区分多NPU）
        # 4. 内存模块（按需分配，不超节点）
        resources:
          requests:
            memory: 8Gi  # 可根据需求调整为16Gi/32Gi
      # 5. 存储卷模块（复用已有的DataVolume和cloud-init）
      volumes:
      - name: datavolumedisk
        dataVolume:
          name: dv-http-focal  # 确保`kubectl get dv -n default`显示Bound
      - name: cloudinitdisk
        cloudInitNoCloud:
          userData: |
            #cloud-config
            password: ubuntu
            chpasswd: { expire: False }  # 密码不失效
            ssh_pwauth: True  # 允许SSH密码登录（测试用）
            users:
            - name: ubuntu
              groups: sudo
              shell: /bin/bash
              sudo: ["ALL=(ALL) NOPASSWD:ALL"]
      # 6. 网络卷模块（匹配网络设备）
      networks:
      - name: default
        pod: {}  # 与interfaces的name对应，使用Pod网络
```



添加标签

```
# 1. 添加 CPU 模型标签（对应 cortex-a57）
kubectl label node kubevirt1910b cpu-model.node.kubevirt.io/cortex-a57=true --overwrite

# 2. 添加机器类型标签（对应 machine.type: virt）
kubectl label node kubevirt1910b machine-type.node.kubevirt.io/virt=true --overwrite
```



创建虚拟机

```
kubectl delete vmi ubuntu20-vm -n default --force --grace-period=0

kubectl delete vm ubuntu20-vm -n default --force --grace-period=0

kubectl apply -f vm.yaml 

virtctl start ubuntu20-vm

kubectl get vm

kubectl get vmi

kubectl get pod 

kubectl describe vm  ubuntu20-vm

kubectl describe vmi  ubuntu20-vm
```
